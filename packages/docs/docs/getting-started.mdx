## Installation

```bash
yarn add @tchayen/design-system
```

## Usage

```jsx
import { Provider } from "@tchayen/design-system";

export default function App() {
  return <Provider>{/* Your app */}</Provider>;
}
```

```jsx
import { MagicScriptTag } from "@tchayen/design-system";

// ...
<MagicScriptTag />;
```

```jsx
<Button
  onPress={() => console.log("I got pressed!")}
  variant="secondary"
  isDisabled
>
  Press me
</Button>
```
